    Create `src/providers/MotionProvider.tsx` as a client component wrapping children in LazyMotion with domAnimation and strict mode:

    ```typescript
    'use client'

    import { LazyMotion, domAnimation } from 'motion/react'

    export default function MotionProvider({
      children,
    }: {
      children: React.ReactNode
    }) {
      return (
        <LazyMotion features={domAnimation} strict>
          {children}
        </LazyMotion>
      )
    }
    ```

    IMPORTANT: Import `LazyMotion` and `domAnimation` from `motion/react` (not `framer-motion`). The `strict` prop ensures any accidental use of `motion.*` instead of `m.*` components throws a runtime error, preventing full bundle imports.

    Create `src/components/ui/AnimatedSection.tsx` as a client component using `m` from `motion/react-m`:

    ```typescript
    'use client'

    import { m } from 'motion/react-m'

    interface AnimatedSectionProps {
      children: React.ReactNode
      className?: string
      delay?: number
    }

    const variants = {
      hidden: { opacity: 0, y: 32 },
      visible: { opacity: 1, y: 0 },
    }

    export default function AnimatedSection({
      children,
      className,
      delay = 0,
    }: AnimatedSectionProps) {
      return (
        <m.div
          className={className}
          initial="hidden"
          whileInView="visible"
          viewport={{ once: true, margin: '-80px' }}
          transition={{ duration: 0.6, delay, ease: [0.22, 1, 0.36, 1] }}
          variants={variants}
        >
          {children}
        </m.div>
      )
    }
    ```

    IMPORTANT: Import `{ m }` from `motion/react-m` (NOT `{ motion }` from `motion/react`). The `m` component is the tree-shakeable version that works with LazyMotion. Using `motion` instead would bypass LazyMotion and ship the full ~34KB bundle. The `strict` prop on LazyMotion will throw a runtime error if this is done wrong.
